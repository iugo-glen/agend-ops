```javascript
  return '$' + amount.toLocaleString(undefined, {minimumFractionDigits: 2, maximumFractionDigits: 2}) + ' ' + (currency || 'AUD');
}
```

```javascript
function daysOverdue(dueDateStr) {
  var due = new Date(dueDateStr + 'T00:00:00');
  var now = new Date();
  now.setHours(0, 0, 0, 0);
  var diff = Math.floor((now - due) / (1000 * 60 * 60 * 24));
  return diff > 0 ? diff : 0;
}
```

e. Add invoice rendering logic after the existing populateColumn calls:

```javascript
/* --- Invoices Column --- */
var invoiceItems = null;
if (invoicesResult) {
  // Filter out written-off, sort: overdue first, then sent, draft, reminder, paid
  var todayStr = new Date().getFullYear() + '-' +
    String(new Date().getMonth() + 1).padStart(2, '0') + '-' +
    String(new Date().getDate()).padStart(2, '0');

  invoiceItems = invoicesResult
    .filter(function(r) { return r.status !== 'written-off'; })
    .map(function(r) {
      // Compute overdue flag
      r._isOverdue = (r.status === 'sent' && r.due_date && r.due_date < todayStr);
      r._daysOverdue = r._isOverdue ? daysOverdue(r.due_date) : 0;
      return r;
    })
    .sort(function(a, b) {
      // Sort order: overdue (by days desc), sent, draft, reminder, paid
      var statusOrder = {'overdue': 0, 'sent': 1, 'draft': 2, 'reminder': 3, 'paid': 4, 'disputed': 1.5, 'written-off': 5};
      var aOrder = a._isOverdue ? 0 : (statusOrder[a.status] || 5);
      var bOrder = b._isOverdue ? 0 : (statusOrder[b.status] || 5);
      if (aOrder !== bOrder) return aOrder - bOrder;
      if (a._isOverdue && b._isOverdue) return b._daysOverdue - a._daysOverdue;
      // Within same status group, sort by due_date ascending
      var aDue = a.due_date || '9999-99-99';
      var bDue = b.due_date || '9999-99-99';
      return aDue < bDue ? -1 : aDue > bDue ? 1 : 0;
    });

  // Add group labels between status groups
  // (handled in createInvoiceCard by checking previous item's effective status)
}

populateColumn('cards-invoices', 'count-invoices', invoiceItems, createInvoiceCard, 'invoices');
```
