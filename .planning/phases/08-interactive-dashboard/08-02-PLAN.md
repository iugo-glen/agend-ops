```typescript
export interface FeedEntry {
  ts: string; type: string; summary: string; level: string; trigger: string;
  details?: Record<string, unknown>; duration_ms?: number;
}
export interface TriageRecord {
  message_id: string; thread_id: string; from: string; from_name?: string;
  subject: string; received: string; priority: string; starred?: boolean;
  snippet?: string; action_type?: string; draft_id?: string;
  client_name?: string; suggested_action?: string;
}
export interface TaskRecord {
  id: string; ts: string; status: string; description: string; trigger: string;
  task_type?: string; client_name?: string; draft_id?: string;
}
export interface TodoRecord {
  id: string; ts: string; status: string; text: string; priority: string;
  due_date?: string | null; category?: string | null; completed_at?: string | null;
}
export interface InvoiceRecord {
  id: string; ts: string; status: string; client_name: string;
  amount?: number | null; currency?: string; due_date?: string | null;
  project_code?: string | null; invoice_number?: string | null;
}
export interface QueuedAction {
  id: string; ts: string;
  action: 'mark-paid' | 'complete-todo' | 'complete-task' | 'trigger-triage';
  target_id: string; status: 'queued' | 'completed' | 'failed';
  requested_by: string;
}
```

API Routes (from Plan 01):
- GET /api/data/[type] -- returns JSON array for: feed, triage, tasks, todos, invoices
- POST /api/actions -- body: { action, target_id }, returns { queued: true, id }
