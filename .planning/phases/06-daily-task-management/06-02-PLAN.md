Populate the Today column with a combined view per D-08:
- First section: active to-dos (section label "YOUR TO-DOS"), sorted by priority (high->normal->low), then due_date
- Second section: pending Claude tasks (section label "CLAUDE TASKS"), reusing createTaskCard
- Use section-label dividers between the two groups

```javascript
/* Today Column */
var todayContainer = document.getElementById('cards-today');
var todayCount = document.getElementById('count-today');
todayContainer.textContent = '';

var activeTodoItems = todosResult
  ? todosResult.filter(function(r) { return r.status === 'active'; })
      .sort(function(a, b) {
        var priOrder = { 'high': 0, 'normal': 1, 'low': 2 };
        var priDiff = (priOrder[a.priority] || 1) - (priOrder[b.priority] || 1);
        if (priDiff !== 0) return priDiff;
        var aDate = a.due_date || '9999-99-99';
        var bDate = b.due_date || '9999-99-99';
        return aDate < bDate ? -1 : aDate > bDate ? 1 : 0;
      })
  : [];

var pendingTaskItems = tasksResult
  ? tasksResult.filter(function(r) { return r.status === 'pending' || r.status === 'in-progress'; })
  : [];

var totalToday = activeTodoItems.length + pendingTaskItems.length;

if (totalToday === 0) {
  todayContainer.appendChild(renderEmptyState('today'));
  todayCount.textContent = '(0)';
} else {
  todayCount.textContent = '(' + totalToday + ')';

  if (activeTodoItems.length > 0) {
    var todoLabel = document.createElement('div');
    todoLabel.className = 'section-label';
    todoLabel.textContent = 'YOUR TO-DOS (' + activeTodoItems.length + ')';
    todayContainer.appendChild(todoLabel);
    activeTodoItems.forEach(function(item) {
      todayContainer.appendChild(createTodoCard(item));
    });
  }

  if (pendingTaskItems.length > 0) {
    var taskLabel = document.createElement('div');
    taskLabel.className = 'section-label';
    taskLabel.textContent = 'CLAUDE TASKS (' + pendingTaskItems.length + ')';
    todayContainer.appendChild(taskLabel);
    pendingTaskItems.forEach(function(item) {
      todayContainer.appendChild(createTaskCard(item));
    });
  }
}
```

**Important:** Maintain all existing XSS safety patterns -- use textContent only, never innerHTML. All dynamic values must be set via textContent or setAttribute, never string interpolation into HTML.
