d. Add briefing banner rendering after the "Last Updated" section:
```javascript
/* --- Briefing Banner --- */
var banner = document.getElementById('briefing-banner');
if (briefingResult && briefingResult !== null && briefingResult.details) {
  var details = briefingResult.details;
  document.getElementById('brief-emails').textContent = details.emails_scanned || 0;
  document.getElementById('brief-urgent').textContent = details.urgent || 0;
  document.getElementById('brief-tasks').textContent = details.pending_tasks || 0;
  document.getElementById('brief-deadlines').textContent = details.deadlines_48h || 0;
  document.getElementById('briefing-date').textContent = briefingResult.details.briefing_file
    ? briefingResult.details.briefing_file.replace('data/briefings/', '').replace('.md', '')
    : '';
  banner.hidden = false;
} else {
  banner.hidden = true;
}
```
